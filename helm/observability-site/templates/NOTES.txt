1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  Ingress configured for:
{{- range .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}{{ range .paths }}{{ .path }}{{ end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get nodes --namespace {{ .Values.namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  export NODE_PORT=$(kubectl get svc --namespace {{ .Values.namespace }} {{ include "observability-site.fullname" . }} -o jsonpath='{.spec.ports[0].nodePort}')
  echo http://$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status of by running 'kubectl get svc -w {{ include "observability-site.fullname" . }} --namespace {{ .Values.namespace }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Values.namespace }} {{ include "observability-site.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Values.namespace }} -l "app.kubernetes.io/name={{ include "observability-site.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Values.namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Values.namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the deployment status:
  kubectl get pods --namespace {{ .Values.namespace }} -l "app.kubernetes.io/name={{ include "observability-site.name" . }}"
  kubectl get deployment --namespace {{ .Values.namespace }}

3. View logs:
  kubectl logs -n {{ .Values.namespace }} deployment/{{ include "observability-site.fullname" . }} -f

4. Watch HPA scaling:
  kubectl get hpa -n {{ .Values.namespace }} -w
